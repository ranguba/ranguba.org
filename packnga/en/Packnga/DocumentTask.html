def initialize(spec)
  @spec = spec
  @yard_task = YARDTask.new(@spec)
  @reference_task = ReferenceTask.new(@spec)
  self.base_dir = "doc"
  yield(self) if block_given?
  define
end

def base_dir=(dir)
