def dump(options={})
  options ||= {}
  if options.is_a?(Hash)
    options = options.dup
    output = options.delete(:output)
  else
    output, options = options, {}
  end
  new(options).dump(output)
end

def dump(output=nil)
  dumped_schema = @schema.dump
  return nil if dumped_schema.nil?

  return_string = false
  if output.nil?
    output = StringIO.new
    return_string = true
  end
  version = @version || management_table.current_version
  output << "ActiveGroonga::Schema.define(:version => #{version}) do |schema|\n"
  output << "  schema.instance_eval do\n"
  dumped_schema.each_line do |line|
    if /^\s*$/ =~ line
      output << line
    else
      output << "    #{line}"
    end
  end
  output << "  end\n"
  output << "end\n"
  if return_string
    output.string
  else
    output
  end
end
