def migrate(direction)
  result = nil
  case direction
  when :up
    report("migrating")
  when :down
    report("reverting")
  end
  time = Benchmark.measure do
    result = send(direction)
  end
  case direction
  when :up
    report("migrated (%.4fs)" % time.real)
  when :down
    report("reverted (%.4fs)" % time.real)
  end
  result
end
