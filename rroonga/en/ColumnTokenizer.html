def tokenize_column_list(column_list)
  tokens = column_list.split(/[\s,]/)
  tokens.reject!(&:empty?)
  tokens.select! do |token|
    token == "*" || token =~ /[A-Za-z0-9_]/
  end
  tokens.each do |token|
    unless token == "*"
      token.sub!(/[^A-Za-z0-9_]\z/, '')
    end
  end
end
