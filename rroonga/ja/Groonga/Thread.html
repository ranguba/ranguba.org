static VALUE
rb_grn_thread_s_set_limit_getter (VALUE self, VALUE rb_getter)
{
    rb_iv_set(self, "@limit_getter", rb_getter);

    if (NIL_P(rb_getter)) {
        grn_thread_set_get_limit_func(NULL, NULL);
    } else {
        grn_thread_set_get_limit_func(rb_grn_thread_get_limit_func,
                                      (void *)rb_getter);
    }

    return Qnil;
}
