def guess_invocation_label(query, method_name, index)
  if method_name.to_s =~ /drilldown|do_group/
    columns = tokenize_column_list(query.drilldown_columns).uniq
    columns[index - 1]
  else
    raise "bad: #{method_name}"
  end
end
