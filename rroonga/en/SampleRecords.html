def load(page)
  @current_count += 1
  record = {
    :_key => page[:title],
    :content => page[:content],
  }
  #pp record
  @records << record
  throw :stop_extract if @current_count == @record_count
end

def values(count=nil)
  count ||= @record_count

  if count == 1
    [first_record]
  else
    @records[0, count - 1] + [first_record]
  end
end
