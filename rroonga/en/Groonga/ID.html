static VALUE
rb_grn_id_s_builtin_p (VALUE self, VALUE rb_id)
{
    grn_ctx *ctx = NULL;
    grn_id id;

    id = NUM2INT(rb_id);

    return CBOOL2RVAL(grn_id_is_builtin(ctx, id));
}
