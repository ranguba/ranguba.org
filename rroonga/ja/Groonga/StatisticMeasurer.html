def measure_disk_usage(path)
  return 0 if path.nil?

  usage = File.size(path)
  1.step(MAX_N_ADDITIONAL_PATHS) do |i|
    additional_path = "%s.%03X" % [path, i]
    break unless File.exist?(additional_path)
    usage += File.size(additional_path)
  end
  usage
end
