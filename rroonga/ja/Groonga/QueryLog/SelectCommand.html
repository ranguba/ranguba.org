def conditions
  @conditions ||= filter.split(/(?:&&|&!|\|\|)/).collect do |condition|
    condition = condition.strip
    condition = condition.gsub(/\A[\s\(]*/, '')
    condition = condition.gsub(/[\s\)]*\z/, '') unless /\(/ =~ condition
    condition
  end
end
