static VALUE
rb_grn_procedure_stable_p (VALUE self)

    grn_ctx *context;
    grn_obj *procedure;
    grn_bool is_stable;

    procedure = RVAL2GRNOBJECT(self, &context);

    return CBOOL2RVAL(is_stable);
