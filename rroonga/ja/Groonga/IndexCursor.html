static VALUE
rb_grn_index_cursor_each (VALUE self)
{
    grn_obj *cursor;
    grn_ctx *context;
    VALUE rb_table;
    VALUE rb_lexicon;

    RETURN_ENUMERATOR(self, 0, NULL);

    rb_grn_index_cursor_deconstruct(SELF(self), &cursor, &context,
				    NULL, NULL, NULL, NULL);

    if (!context) {
	return Qnil;
    }

    if (!cursor) {
	return Qnil;
    }

    rb_table = rb_iv_get(self, "@table");
    rb_lexicon = rb_iv_get(self, "@lexicon");
    while (GRN_TRUE) {
	VALUE rb_posting;
	rb_posting = next_value(context, cursor, rb_table, rb_lexicon);
	if (NIL_P(rb_posting)) {
	    break;
	}
	rb_yield(rb_posting);
    }

    return Qnil;
}

static VALUE
rb_grn_index_cursor_next (VALUE self)
{
    VALUE rb_posting = Qnil;
    grn_obj *cursor;
    grn_ctx *context;

    rb_grn_index_cursor_deconstruct(SELF(self), &cursor, &context,
				    NULL, NULL, NULL, NULL);
    if (context && cursor) {
	VALUE rb_table;
	VALUE rb_lexicon;
	rb_table = rb_iv_get(self, "@table");
	rb_lexicon = rb_iv_get(self, "@lexicon");
	rb_posting = next_value(context, cursor, rb_table, rb_lexicon);
    }

    return rb_posting;

}
