def create!(attributes=nil, &block)
  if attributes.is_a?(Array)
    attributes.collect do |nested_attributes|
      create!(nested_attributes, &block)
    end
  else
    object = new(attributes)
    yield(object) if block_given?
    object.save!
    object
  end
end
