static VALUE
rb_grn_name_s_column_p (VALUE self, VALUE rb_name)
{
    grn_ctx *ctx = NULL;
    const char *name;
    size_t name_size;

    name = StringValueCStr(rb_name);
    name_size = RSTRING_LEN(rb_name);

    return CBOOL2RVAL(grn_obj_name_is_column(ctx, name, name_size));
}
